# Cron schedule for automated backups
# Add this to crontab: crontab -e

# Daily backup at 2 AM
0 2 * * * /path/to/aurumvault/scripts/backup-database.sh >> /var/log/aurumvault-backup.log 2>&1

# Weekly full backup on Sunday at 3 AM
0 3 * * 0 /path/to/aurumvault/scripts/backup-database.sh >> /var/log/aurumvault-backup-weekly.log 2>&1

# Monthly backup on 1st of month at 4 AM
0 4 1 * * /path/to/aurumvault/scripts/backup-database.sh >> /var/log/aurumvault-backup-monthly.log 2>&1

# Backup verification every day at 6 AM
0 6 * * * /path/to/aurumvault/scripts/verify-backups.sh >> /var/log/aurumvault-verify.log 2>&1
